    <script>
        // JS 逻辑核心保持一致，优化状态更新
        const state = {
            isDragging: false,
            startX: 0, startY: 0,
            cropRect: { x: 0, y: 0, w: 0, h: 0 },
            capturedFrames: [],
            keyColor: [0, 255, 0],
            isProcessing: false
        };

        const elements = {
            video: document.getElementById('mainVideo'),
            videoWrapper: document.getElementById('videoWrapper'),
            cropBox: document.getElementById('cropBox'),
            mainPreview: document.getElementById('mainPreview'),
            finalCanvas: document.getElementById('finalCanvas'),
            btnGenerate: document.getElementById('btnGenerate'),
            status: document.getElementById('globalStatus'),
            btnText: document.getElementById('btnText')
        };

        const pCtx = elements.mainPreview.getContext('2d', { willReadFrequently: true });

        document.getElementById('videoInput').onchange = (e) => {
            const file = e.target.files[0];
            if (file) {
                elements.video.src = URL.createObjectURL(file);
                elements.video.play();
                elements.status.textContent = "已就绪：请划定裁剪区域";
                elements.status.className = "mt-4 md:mt-0 text-sm font-medium px-4 py-2 rounded-full bg-blue-100 text-blue-700";
                document.getElementById('stepMatting').classList.remove('step-inactive');
            }
        };

        elements.videoWrapper.onmousedown = (e) => {
            state.isDragging = true;
            const r = elements.videoWrapper.getBoundingClientRect();
            state.startX = e.clientX - r.left;
            state.startY = e.clientY - r.top;
            elements.cropBox.style.display = 'block';
            updateCropBox(state.startX, state.startY, state.startX, state.startY);
        };

        window.onmousemove = (e) => {
            if (!state.isDragging) return;
            const r = elements.videoWrapper.getBoundingClientRect();
            let curX = Math.max(0, Math.min(e.clientX - r.left, r.width));
            let curY = Math.max(0, Math.min(e.clientY - r.top, r.height));
            updateCropBox(state.startX, state.startY, curX, curY);
        };

        window.onmouseup = () => state.isDragging = false;

        function updateCropBox(x1, y1, x2, y2) {
            state.cropRect = {
                x: Math.min(x1, x2), y: Math.min(y1, y2),
                w: Math.abs(x1 - x2), h: Math.abs(y1 - y2)
            };
            elements.cropBox.style.left = state.cropRect.x + 'px';
            elements.cropBox.style.top = state.cropRect.y + 'px';
            elements.cropBox.style.width = state.cropRect.w + 'px';
            elements.cropBox.style.height = state.cropRect.h + 'px';
        }

        function processFrame(ctx, w, h) {
            const imageData = ctx.getImageData(0, 0, w, h);
            const data = imageData.data;
            const [kr, kg, kb] = state.keyColor;
            const tol = parseInt(document.getElementById('inputTol').value);
            const feather = parseFloat(document.getElementById('inputFeather').value);
            const doSpill = document.getElementById('checkSpill').checked;
            const mode = document.getElementById('previewMode').value;

            for (let i = 0; i < data.length; i += 4) {
                const r = data[i], g = data[i+1], b = data[i+2];
                const dist = Math.sqrt((r-kr)**2 + (g-kg)**2 + (b-kb)**2);
                
                let alpha = 255;
                if (dist < tol) alpha = 0;
                else if (dist < tol + (feather * 10)) alpha = ((dist - tol) / (feather * 10)) * 255;

                if (doSpill && alpha > 0 && kg > 100) {
                    data[i+1] = Math.min(g, (r + b) / 2);
                }

                if (mode === 'alpha') {
                    data[i] = data[i+1] = data[i+2] = alpha;
                    data[i+3] = 255;
                } else if (mode === 'white') {
                    const a = alpha / 255;
                    data[i] = r * a + 255 * (1-a);
                    data[i+1] = g * a + 255 * (1-a);
                    data[i+2] = b * a + 255 * (1-a);
                    data[i+3] = 255;
                } else if (mode === 'black') {
                    const a = alpha / 255;
                    data[i] = r * a; data[i+1] = g * a; data[i+2] = b * a;
                    data[i+3] = 255;
                } else {
                    data[i+3] = alpha;
                }
            }
            ctx.putImageData(imageData, 0, 0);
        }

        document.getElementById('keyColor').oninput = (e) => {
            const hex = e.target.value;
            state.keyColor = [parseInt(hex.slice(1,3), 16), parseInt(hex.slice(3,5), 16), parseInt(hex.slice(5,7), 16)];
        };

        document.getElementById('inputTol').oninput = (e) => document.getElementById('valTol').textContent = e.target.value;
        document.getElementById('inputFeather').oninput = (e) => document.getElementById('valFeather').textContent = e.target.value;
        document.getElementById('inputFPS').oninput = (e) => document.getElementById('valFPS').textContent = e.target.value;

        function renderLoop() {
            if (state.cropRect.w > 5 && state.cropRect.h > 5) {
                const sX = elements.video.videoWidth / elements.videoWrapper.clientWidth;
                const sY = elements.video.videoHeight / elements.videoWrapper.clientHeight;
                const rW = state.cropRect.w * sX, rH = state.cropRect.h * sY;
                elements.mainPreview.width = rW;
                elements.mainPreview.height = rH;
                pCtx.drawImage(elements.video, state.cropRect.x * sX, state.cropRect.y * sY, rW, rH, 0, 0, rW, rH);
                processFrame(pCtx, rW, rH);
                document.getElementById('sizeInfo').textContent = `${Math.round(rW)} x ${Math.round(rH)}`;
            }
            requestAnimationFrame(renderLoop);
        }
        renderLoop();

        elements.btnGenerate.onclick = async () => {
            if (state.cropRect.w < 5) return alert("请先划定裁剪区域");
            state.isProcessing = true;
            elements.btnGenerate.disabled = true;
            elements.btnText.textContent = "正在逐帧提取...";
            state.capturedFrames = [];

            const fps = parseInt(document.getElementById('inputFPS').value);
            const totalFrames = Math.floor(elements.video.duration * fps);
            const interval = 1 / fps;
            elements.video.pause();

            const sX = elements.video.videoWidth / elements.videoWrapper.clientWidth;
            const sY = elements.video.videoHeight / elements.videoWrapper.clientHeight;
            const rW = state.cropRect.w * sX, rH = state.cropRect.h * sY;

            for (let i = 0; i < totalFrames; i++) {
                elements.video.currentTime = i * interval;
                await new Promise(r => elements.video.onseeked = r);
                const offCanvas = document.createElement('canvas');
                offCanvas.width = rW; offCanvas.height = rH;
                const offCtx = offCanvas.getContext('2d');
                offCtx.drawImage(elements.video, state.cropRect.x * sX, state.cropRect.y * sY, rW, rH, 0, 0, rW, rH);
                processFrame(offCtx, rW, rH);
                state.capturedFrames.push(offCanvas.toDataURL('image/png'));
                elements.btnText.textContent = `提取中 (${Math.round((i/totalFrames)*100)}%)`;
            }

            elements.btnText.textContent = "重新提取";
            elements.btnGenerate.disabled = false;
            elements.status.textContent = `完成：已提取 ${totalFrames} 帧`;
            elements.status.className = "mt-4 md:mt-0 text-sm font-medium px-4 py-2 rounded-full bg-green-100 text-green-700";
            document.getElementById('stepExport').classList.remove('step-inactive');
            elements.video.play();
        };

        function generateFinalSprite() {
            const frames = state.capturedFrames;
            const cols = parseInt(document.getElementById('exportCols').value);
            const scale = parseFloat(document.getElementById('exportScale').value);
            const rows = Math.ceil(frames.length / cols);
            if (frames.length === 0) return null;
            return new Promise(resolve => {
                const tempImg = new Image();
                tempImg.onload = () => {
                    const fw = tempImg.width * scale;
                    const fh = tempImg.height * scale;
                    elements.finalCanvas.width = fw * cols;
                    elements.finalCanvas.height = fh * rows;
                    const ctx = elements.finalCanvas.getContext('2d');
                    let loaded = 0;
                    frames.forEach((src, idx) => {
                        const img = new Image();
                        img.onload = () => {
                            ctx.drawImage(img, (idx % cols) * fw, Math.floor(idx / cols) * fh, fw, fh);
                            if (++loaded === frames.length) resolve(elements.finalCanvas);
                        };
                        img.src = src;
                    });
                };
                tempImg.src = frames[0];
            });
        }

        document.getElementById('btnPreviewSprite').onclick = async () => {
            await generateFinalSprite();
            document.getElementById('modal').style.display = 'flex';
        };

        document.getElementById('btnDownloadPNG').onclick = async () => {
            const canvas = await generateFinalSprite();
            const link = document.createElement('a');
            link.download = `sprite_${Date.now()}.png`;
            link.href = canvas.toDataURL('image/png');
            link.click();
        };

        document.getElementById('btnDownloadZIP').onclick = async () => {
            const zip = new JSZip();
            state.capturedFrames.forEach((data, i) => zip.file(`f_${String(i).padStart(3, '0')}.png`, data.split(',')[1], {base64: true}));
            const content = await zip.generateAsync({type:"blob"});
            const link = document.createElement('a');
            link.href = URL.createObjectURL(content);
            link.download = "frames.zip";
            link.click();
        };

        function closeModal() { document.getElementById('modal').style.display = 'none'; }
    </script>
